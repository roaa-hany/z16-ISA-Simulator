.text
.org 0
main:
    li sp, 0xFF     # Initialize stack pointer
    li a0, 42       # Load value 42
    push a0         # Store on stack
    
    li a1, 99       # Load another value 99
    push a1         # Store on stack

    pop a2          # Retrieve first value (99)
    pop a3          # Retrieve second value (42)

    add a0, a2, a3  # Add retrieved values (99 + 42)
    ecall 1         # Print sum

exit:
    ecall 3         # Exit program
