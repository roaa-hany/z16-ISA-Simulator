.text
.org 0
main:
    li a0, 10       # Counter
    li a1, 3        # Decrement value

loop:
    beq a0, zero, exit # Exit loop if counter == 0
    sub a0, a0, a1  # a0 = a0 - 3
    ecall 1         # Print a0 after each step
    j loop          # Repeat loop

exit:
    ecall 3         # Exit program
