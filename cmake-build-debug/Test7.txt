.text
.org 0
main:
    li a0, 64       # Load immediate 64
    li a1, 2        # Load shift amount 2
    srl a2, a0, a1  # Logical right shift (64 >> 2)
    sll a3, a0, a1  # Logical left shift (64 << 2)

    bgt a3, a2, greater_label # If (a3 > a2), branch

    li a0, -1       # If branch not taken, set -1
    ecall 1         # Print result
    j exit

greater_label:
    li a0, 1        # If branch taken, set 1
    ecall 1         # Print result

exit:
    ecall 3         # Exit program
