.org 0
main:
    li      a0, 0xFF     # Load 0xFF into a0
    li      a1, 0xA5     # Load 0xA5 into a1
    not     t0, a0       # t0 = ~a0
    andi    t1, a0, 0x0F # t1 = a0 & 0x0F
    ori     s0, a1, 0xF0 # s0 = a1 | 0xF0
    xori    s1, a1, 0xFF # s1 = a1 ^ 0xFF
    ecall   3            # Terminate program
