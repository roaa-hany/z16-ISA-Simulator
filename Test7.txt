.text
.org 0
main:
    li a0, 5       # Load immediate 5
    li a1, 2        # Load shift amount 2
    srl t0, a0      # Logical right shift (5 >> 2)
    sll t1, a0      # Logical left shift (5 << 2)

    blt t0, t1, greater_label # If (t1 > t0), branch

    li a0, -1       # If branch not taken, set -1
    ecall 1         # Print result
    j exit

greater_label:
    li a0, 1        # If branch taken, set 1
    ecall 1         # Print result

exit:
    ecall 3         # Exit program
